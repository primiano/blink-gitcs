Tests that setting an outline or a border of 0 on an element without a border or outline doesn't repaint.

(GraphicsLayer
  (bounds 800.00 600.00)
  (children 1
    (GraphicsLayer
      (bounds 800.00 600.00)
      (contentsOpaque 1)
      (drawsContent 1)
      (repaint rects
        (rect 10.00 10.00 200.00 200.00)
        (rect 10.00 10.00 100.00 100.00)
      )
    )
  )
)
