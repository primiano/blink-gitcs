This test checks that we correctly invalidate a shifted fixed positioned element overlapping another fixed positioned element.

For this test, there should be only one green square below.

{
  "bounds": [800, 600],
  "children": [
    {
      "bounds": [800, 600],
      "contentsOpaque": true,
      "drawsContent": true,
      "children": [
        {
          "children": [
            {
              "position": [8, 88]
            },
            {
              "position": [8, 88],
              "bounds": [800, 600],
              "drawsContent": true,
              "repaintRects": [
                [142, 62, 100, 100],
                [142, 62, 100, 100],
                [0, 0, 100, 100]
              ]
            }
          ]
        }
      ]
    }
  ]
}
